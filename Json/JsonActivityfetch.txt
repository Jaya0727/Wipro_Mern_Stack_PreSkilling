<body>

<h3>Users Todo Tasks</h3>
<ul id="todos"></ul>

<script>
  function fetchingTodos() {
    return new Promise((resolve, reject) => {
      fetch('https://jsonplaceholder.typicode.com/todos')
        .then(response => {
          if (!response.ok) {
            reject("Failed to fetch todos");
          }
          return response.json();
        })
        .then(data => resolve(data))
        .catch(err => reject(err));
    });
  }

  fetchingTodos()
    .then(todos => {
      const ul = document.getElementById("todos");

      // Show only first 5 users tasks
      const firstFive = todos.slice(0, 5);

      firstFive.forEach(todo => {
        const li = document.createElement("li");
        li.textContent = `User ID: ${todo.userId}, Email: Not Available, Completed: ${todo.completed}`;
        ul.append(li);
      });
    })
    .catch(err => {
      console.error("Error:", err);
      document.body.innerHTML += `<p style="color:red;">${err}</p>`;
    });
</script>

</body>
</html>
